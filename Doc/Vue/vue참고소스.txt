  linkHandler(event: any, path: string) {
    let currentPosY = window.scrollY; // 초기 시작점
    let getItem = window.sessionStorage.getItem('pagePos');
    if (getItem) window.sessionStorage.removeItem('pagePos');
    window.sessionStorage.setItem('pagePos', JSON.stringify({
      key: 'myCommunityBoardItem',
      posY: currentPosY,
      index: this.$vnode.key
    }));
    this.$router.push(path);
  }
